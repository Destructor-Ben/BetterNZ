### Economy Development Focus ###
# The focus tooltip
BNZ_develop_polynesian_economy_tooltip_effect = {
	random_state = {
		limit = {
			has_state_flag = NZL_BNZ_develop_polynesian_economy
		}

		BNZ_develop_polynesian_economy = yes
	}
}

# The focus reward
BNZ_develop_polynesian_economy_effect = {
	random_owned_controlled_state = {
		prioritize = {
			# BNZ-APS (minus north and south island)
			# 284 # North Island
			# 723 # South Island
			726 # Samoa
			629 # Hawaii
			684 # Caroline Islands
			638 # Guam
			641 # Tahiti
			646 # Saipan
			
			524 # Taiwan
			630 # Johnston Atoll
			631 # Midway Island
			632 # Wake Island
			633 # Marshall Islands
			634 # Solomon Islands
			635 # New Caledonia
			636 # Fiji
			639 # Gilbert Islands
			642 # Phoenix Island
			643 # Ellice Islands
			645 # Iwo Jima
			647 # Palau
			648 # Marcus Island
			725 # Nauru
			727 # Line Islands
			734 # Nendo
			270 # Pitcairn
			948 # Easter Island
		}

		limit = {
			free_building_slots = {
				building = infrastructure
				size > 1
			}
		}

		BNZ_develop_polynesian_economy = yes

		set_state_flag = NZL_BNZ_develop_polynesian_economy
	}
}

# The actual effect
BNZ_develop_polynesian_economy = {
	add_building_construction = {
		type = infrastructure
		level = 2
		instant_build = yes
	}
	
	add_extra_state_shared_building_slots = 1
}
